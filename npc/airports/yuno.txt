y_airport,143,43,5	script	Equipe do Dirigível#y_air1a::y_airport1	90,{
	mes "[Equipe do Dirigível]";
	mes "Bem-vindo ao Aeroporto de Juno, onde oferecemos voos domésticos para Einbroch, Lighthalzen e Hugel,";
	mes "e voos internacionais para Izlude e Rachel.";
	mes "Como posso ser útil?";
	next;
	if (select("Embarcar no Dirigível:Cancelar.") == 1) {
		mes "[Equipe do Dirigível]";
		mes "A taxa de embarque para todos";
		mes "os voos é de 1.200 zeny. Se você";
		mes "usar um Ticket Gratuito para Dirigível,";
		mes "a taxa de embarque será dispensada. Então, você gostaria de partir?";
		next;
		if (select("Sim:Não") == 1) {
			if (countitem(7311) > 0) {
				delitem 7311,1; //Free_Flying_Ship_Ticket
				warp "y_airport",148,51;
				end;
			}
			if (Zeny >= 1200) {
				set Zeny, Zeny - 1200;
				warp "y_airport",148,51;
				end;
			}
			mes "[Equipe do Dirigível]";
			mes "Me desculpe, mas você não";
			mes "tem 1,200 zeny para pagar";
			mes "a taxa de embarque.";
			close;
		}
	}
	mes "[Equipe do Dirigível]";
	mes "Obrigado e";
	mes "Tenha um bom dia~";
	close;
}
