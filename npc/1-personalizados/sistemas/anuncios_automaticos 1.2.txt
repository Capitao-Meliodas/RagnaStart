-	script	News	-1,{
// Timer 1 em 1 hora.
OnMinute30:
	setarray .cores$[0],"0x99CC00","0x00FFFF","0xFFCC99","0xCCFFFF","0xFFFFFF","0xCC99FF","0xFF0000","0xADD8E6"; // Cores dos Anuncios
	setarray .News$[0],	// Random Mensagens
		"[ Servidor ] : Nós da staff agradecemos a sua presença em nosso Servidor !",
		"[ Servidor ] : Nós da staff nunca pediremos seu login ou senha, fique atento !",
		"[ Servidor ] : Não passe seu login ou senha para ninguém, fique seguro.",
		"[ Servidor ] : Os vencedores do TOP Ranking PVP ganharão 50k de Rops no Final de cada Mês !",
		"[ Servidor ] : Eventos serão realizados pela staff todos os Domingos !",
		"[ Servidor ] : Servidor Atualizado, Avante Servidor !",
		"[ Servidor ] : Fique ligado em nossa comunidade online... www.Servidor.com/forum ~ !";
		
	announce .News$[ rand( getarraysize( .News$ ) ) ],bc_all,.cores$[ rand( getarraysize( .cores$ ) ) ]; // Anuncio.
	end;
}
