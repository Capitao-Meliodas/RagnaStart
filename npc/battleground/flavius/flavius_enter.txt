bat_room,133,178,5	script	Oficial Flavius#01a	418,{
	if (checkweight(1201,1) == 0) {
		mes "- Espere um minuto !! -";
		mes "- Atualmente você esta carregando -";
		mes "- muitos itens com você. -";
		mes "- Por favor, tente novamente -";
		mes "- assim que você armazenar alguns itens. -";
		close;
	}
	mes "[Oficial do Exercito Guillaume]";
	mes "Vamos mostrar o poder do excercito Guillaume aos membro do Croixs!";
	next;
	switch(select("Eu quero me juntar a tropa!:Terminar conversa.")) {
	case 1:
		if ((Class == Job_Novice) || (BaseClass == Job_Super_Novice)) {
			mes "[Oficial do Exercito Guillaume]";
			mes "Estou orgulhoso que você queira se juntar a nós e lutar contra a tropa Croix, mas infelizmente não posso mandar novatos ao campo de batalha para morrerem";
		}
		else {
			if (BaseLevel < 80) {
				mes "[Oficial do Exercito Guillaume]";
				mes "Estou orgulhoso que você queira se juntar a nós e lutar contra a tropa Croix, mas infelizmente não posso mandar novatos ao campo de batalha para morrerem";
			}
			else {
				set .@chk_urtime,checkquest(2070,PLAYTIME);
				if ((.@chk_urtime == 0)|| (.@chk_urtime == 1)) {
					mes "[Oficial do Exercito Guillaume]";
					mes "Parece que você acabou de voltar do campo de batalha.";
					mes "É muito cedo para você voltar. Vó descansar, e deixe que a gente destrua os inimigos!";
				}
				else {
					if (.@chk_urtime == 2)
						erasequest 2070;
					set .@mapcount2,getmapusers("bat_b01");
					if (.@mapcount2 > 0) {
						mes "[Oficial do Ex?rcito Guillaume]";
						mes "Eu recebi uma informação que um grupo de elite já foi mandando para enfrentar o exercito Croix.";
						mes "Porque você não espera um pouco? Sugiro que você afie suas armas e prepare seus suprimentos para a batalha";
					}
					else {
						if ($@FlaviusBG1 > 0) {
							mes "[Oficial do Exercito Guillaume]";
							mes "Um grupo de elite já esta esperando para ser mandado para o campo de batalha.";
							mes "Porque você não espera um pouco? Sugiro que você afie suas armas e prepare seus suprimentos para a batalha";
						}
						else {
							mes "[Oficial do Exercito Guillaume]";
							mes "Você com certeza parece pronto para a batalha!";
							mes "Vó mostrar aos Croix como guerreiros de verdade lutam!";
							mes "Hoje nosso grito de batalha vai ecoar pelo campo de batalha!";
							close2;
							warp "bat_room",85,223;
							end;
						}
					}
				}
			}
		}
		break;
	case 2:
		mes "[Oficial do Exercito Guillaume]";
		mes "Hoje seremos vitoriosos!";
		break;
	}
	close;
}
